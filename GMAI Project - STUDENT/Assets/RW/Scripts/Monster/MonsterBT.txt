tree("Root")
	fallback
		while not IsDead
			fallback
				tree("Flee")
				tree("Attack")
				tree("Chase")
				tree("Observe")
				tree("Patrol")
				Idle
		Die


tree("Flee")
	sequence
		IsPlayerVisible
		IsLow
		FleePlayer

tree("Attack")
	sequence
		IsPlayerInTerritory
		IsPlayerInAttackRange
		GoToPlayer
		Attack

tree("Patrol")
	while not IsPlayerVisible
		while not IsChasing
			Patrol

tree("Chase")
	sequence
		IsPlayerVisible
		IsPlayerInTerritory
		ChasePlayer

tree("Observe")
	while IsPlayerVisible
		sequence
			IsPlayerVisible
			ObservePlayer
			IsMonsterInTerritory
			GoToPlayer
