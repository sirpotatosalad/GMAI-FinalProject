tree("Root")
	fallback
		while not IsDead
			fallback
				tree("Flee")
				tree("Attack")
				tree("Chase")
				tree("Patrol")
				Idle
		Die


tree("Flee")
	sequence
		IsPlayerVisible
		IsLow
		FleePlayer

tree("Attack")
	sequence
		IsPlayerInAttackRange
		GoToPlayer
		Attack

tree("Patrol")
	while not IsPlayerVisible
		Patrol

tree("Chase")
	while IsPlayerVisible
		ChasePlayer
